#!/bin/bash
# =============================================================================
# OPi2 (Store) — Full Setup Script
# Fresh Ubuntu → ready node in one command
#
# Usage (run as root on OPi2 after fresh Ubuntu install):
#   curl -sL https://devcrm.hltrn.cc/uploads/spaces/11/awg/setup-opi2.sh | sudo bash
#
# What it does:
#   1. WireGuard VPN (10.66.66.14) → VPS <VPN_IP>:51821
#   2. SSH keys + root login
#   3. Autossh reverse tunnel (VPS:2223 → OPi2:22)
#   4. microsocks SOCKS5 on :1080
#   5. sing-box geo-routing
#   6. Store schedule: microsocks active 22:00-09:00 MSK
# =============================================================================

set -euo pipefail

# ── Config ──────────────────────────────────────────────────────────────────
NODE_NAME="opi2-store"
WG_IP="10.66.66.14"
WG_PRIVKEY="WFBEvMzNiBCOReEoLHOrTcpMcyXz6/8n3BzWnPpg9Vg="
VPS_HOST="<VPN_IP>"
VPS_WG_PORT="51821"
VPS_PUBKEY="MkOvu1mRqPd61M5mSlLYaEg5iNxXioknJMSfWWvyi1o="
REVERSE_SSH_PORT="2223"
SOCKS_PORT="1080"
SING_BOX_VERSION="1.11.0"

# SSH keys that get root access
VPS_SSH_KEY="ssh-ed25519 AAAAC3NzaC1lZDI1NTE5AAAAIFd2mIukmButN68b3J7C52frtRVURddVW6MKb8eWd0/+ level-server"
CRM_SSH_KEY="ssh-ed25519 AAAAC3NzaC1lZDI1NTE5AAAAIBppUKrYwTExkgChP1LA+is99MWsFotoquCnF2rgtUfd root@v527836.hosted-by-vdsina.com"

# VPS host key (for known_hosts — no interactive prompts)
VPS_HOST_KEY="<VPN_IP> ssh-ed25519 AAAAC3NzaC1lZDI1NTE5AAAAIHRxgXOqcWHGUNadFJbVZ+g9yRcN012y2B+5S7XDDb8a"

RED='\033[0;31m'
GREEN='\033[0;32m'
YELLOW='\033[1;33m'
NC='\033[0m'

log()  { echo -e "${GREEN}[+]${NC} $*"; }
warn() { echo -e "${YELLOW}[!]${NC} $*"; }
err()  { echo -e "${RED}[✗]${NC} $*"; exit 1; }

# ── Check root ──────────────────────────────────────────────────────────────
[[ $EUID -ne 0 ]] && err "Run as root: sudo bash $0"

log "=== OPi2 Store Node Setup ==="

# ── 0. Fix flash-kernel & amneziawg (known OPi5 issues) ────────────────────
log "Pre-fixing known OPi5 issues..."
export DEBIAN_FRONTEND=noninteractive

# Create dirs flash-kernel expects
mkdir -p /boot/firmware/dtbs

# Remove amneziawg if installed (breaks dpkg on 5.10 kernel)
dpkg --remove --force-depends amneziawg amneziawg-dkms amneziawg-tools 2>/dev/null || true

# Hold flash-kernel to prevent it from breaking apt operations
apt-mark hold flash-kernel 2>/dev/null || true

# Fix any broken dpkg state
dpkg --configure -a 2>/dev/null || true

# ── 1. System packages ─────────────────────────────────────────────────────
log "Installing packages..."
apt-get update -qq 2>/dev/null || true
apt-get install -y -qq \
  wireguard-tools curl wget git iptables net-tools \
  autossh build-essential gcc make 2>/dev/null || {
  warn "Some packages failed, trying individually..."
  for pkg in wireguard-tools curl wget git iptables net-tools autossh build-essential gcc make; do
    apt-get install -y -qq $pkg 2>/dev/null || warn "Failed: $pkg"
  done
}

# ── 2. SSH setup ────────────────────────────────────────────────────────────
log "Configuring SSH..."
mkdir -p /root/.ssh
chmod 700 /root/.ssh

# Add authorized keys
for key in "$VPS_SSH_KEY" "$CRM_SSH_KEY"; do
  grep -qF "$key" /root/.ssh/authorized_keys 2>/dev/null || \
    echo "$key" >> /root/.ssh/authorized_keys
done
chmod 600 /root/.ssh/authorized_keys

# Add VPS to known_hosts (prevents interactive prompt)
grep -qF "<VPN_IP>" /root/.ssh/known_hosts 2>/dev/null || \
  echo "$VPS_HOST_KEY" >> /root/.ssh/known_hosts

# Enable root login via key
sed -i 's/^#*PermitRootLogin.*/PermitRootLogin prohibit-password/' /etc/ssh/sshd_config
sed -i 's/^#*PubkeyAuthentication.*/PubkeyAuthentication yes/' /etc/ssh/sshd_config
systemctl restart sshd 2>/dev/null || systemctl restart ssh 2>/dev/null || true

# Generate root SSH keypair (for connecting TO VPS)
if [[ ! -f /root/.ssh/id_ed25519 ]]; then
  ssh-keygen -t ed25519 -f /root/.ssh/id_ed25519 -N "" -q
  log "Generated SSH keypair"
fi
log "Root pubkey: $(cat /root/.ssh/id_ed25519.pub)"

# ── 3. WireGuard (regular, NOT AmneziaWG) ──────────────────────────────────
log "Configuring WireGuard..."

# Load module
modprobe wireguard || err "WireGuard kernel module not available!"
echo "wireguard" > /etc/modules-load.d/wireguard.conf

# Stop any existing WG/AWG
systemctl stop awg-quick@awg0 2>/dev/null || true
systemctl disable awg-quick@awg0 2>/dev/null || true
wg-quick down wg0 2>/dev/null || true
ip link delete wg0 2>/dev/null || true
ip link delete awg0 2>/dev/null || true

# Write config
mkdir -p /etc/wireguard
cat > /etc/wireguard/wg0.conf << WGEOF
[Interface]
Address = ${WG_IP}/24
PrivateKey = ${WG_PRIVKEY}
MTU = 1280

[Peer]
PublicKey = ${VPS_PUBKEY}
Endpoint = ${VPS_HOST}:${VPS_WG_PORT}
AllowedIPs = 10.66.66.0/24
PersistentKeepalive = 25
WGEOF
chmod 600 /etc/wireguard/wg0.conf

# Start WG
systemctl enable wg-quick@wg0
systemctl start wg-quick@wg0
sleep 2

if wg show wg0 2>/dev/null | grep -q "latest handshake"; then
  log "WireGuard UP — connected to VPS"
else
  warn "WireGuard started, waiting for handshake..."
  sleep 5
  wg show wg0 2>/dev/null || warn "WG may need a moment to connect"
fi

# ── 4. Autossh reverse tunnel ──────────────────────────────────────────────
log "Setting up reverse SSH tunnel (VPS:${REVERSE_SSH_PORT} → localhost:22)..."

cat > /etc/systemd/system/autossh-tunnel.service << TUNNELEOF
[Unit]
Description=AutoSSH reverse tunnel to VPS (port ${REVERSE_SSH_PORT})
After=network-online.target wg-quick@wg0.service
Wants=network-online.target

[Service]
Type=simple
Environment="AUTOSSH_GATETIME=0"
ExecStart=/usr/bin/autossh -M 0 -N \
  -R ${REVERSE_SSH_PORT}:localhost:22 \
  -o StrictHostKeyChecking=no \
  -o ServerAliveInterval=30 \
  -o ServerAliveCountMax=3 \
  -o ExitOnForwardFailure=yes \
  -o ConnectTimeout=10 \
  root@${VPS_HOST}
Restart=always
RestartSec=10

[Install]
WantedBy=multi-user.target
TUNNELEOF

systemctl daemon-reload
systemctl enable autossh-tunnel
systemctl start autossh-tunnel || warn "autossh-tunnel failed to start (VPS may not have our pubkey yet)"

# ── 5. microsocks (SOCKS5) ─────────────────────────────────────────────────
log "Installing microsocks..."
if ! command -v microsocks &>/dev/null; then
  cd /tmp
  rm -rf microsocks
  git clone --depth 1 https://github.com/rofl0r/microsocks.git
  cd microsocks
  make && cp microsocks /usr/local/bin/
  cd /root
fi

cat > /etc/systemd/system/microsocks.service << 'MSEOF'
[Unit]
Description=MicroSOCKS proxy
After=wg-quick@wg0.service
Wants=wg-quick@wg0.service

[Service]
Type=simple
ExecStart=/usr/local/bin/microsocks -i 0.0.0.0 -p 1080
Restart=always
RestartSec=5

[Install]
WantedBy=multi-user.target
MSEOF

systemctl daemon-reload
systemctl enable microsocks
systemctl restart microsocks
log "microsocks running on port ${SOCKS_PORT}"

# ── 6. sing-box ────────────────────────────────────────────────────────────
log "Installing sing-box..."
ARCH=$(dpkg --print-architecture)
case "$ARCH" in
  arm64|aarch64) SB_ARCH="arm64" ;;
  amd64|x86_64)  SB_ARCH="amd64" ;;
  armhf)         SB_ARCH="armv7" ;;
  *)             err "Unsupported arch: $ARCH" ;;
esac

if ! command -v sing-box &>/dev/null; then
  cd /tmp
  wget -q "https://github.com/SagerNet/sing-box/releases/download/v${SING_BOX_VERSION}/sing-box-${SING_BOX_VERSION}-linux-${SB_ARCH}.tar.gz" -O sing-box.tar.gz
  tar xzf sing-box.tar.gz
  cp sing-box-${SING_BOX_VERSION}-linux-${SB_ARCH}/sing-box /usr/local/bin/
  chmod +x /usr/local/bin/sing-box
  cd /root
fi

mkdir -p /etc/sing-box
cat > /etc/sing-box/config.json << 'SBEOF'
{
  "log": {
    "level": "warn",
    "timestamp": true
  },
  "dns": {
    "servers": [
      {
        "type": "udp",
        "tag": "dns-ru",
        "server": "77.88.8.8",
        "server_port": 53
      },
      {
        "type": "udp",
        "tag": "dns-default",
        "server": "1.1.1.1",
        "server_port": 53
      }
    ],
    "rules": [
      {
        "rule_set": ["geosite-ru"],
        "server": "dns-ru"
      }
    ],
    "final": "dns-default",
    "independent_cache": true
  },
  "inbounds": [
    {
      "type": "tun",

      "inet4_address": "172.19.0.1/30",
      "auto_route": true,
      "strict_route": true,
      "sniff": true,
      "exclude_interface": ["wg0"]
    }
  ],
  "outbounds": [

        "update_interval": "120h0m0s"
      },
      {
        "type": "remote",
        "tag": "geosite-ru",
        "format": "binary",
        "url": "https://raw.githubusercontent.com/hiddify/hiddify-geo/rule-set/country/geosite-ru.srs",
        "update_interval": "120h0m0s"

    "final": "direct",
    "auto_detect_interface": true
  }

SBEOF

cat > /etc/systemd/system/sing-box.service << 'SSEOF'
[Unit]
Description=sing-box service
After=network.target wg-quick@wg0.service
Wants=wg-quick@wg0.service

[Service]
Type=simple
ExecStart=/usr/local/bin/sing-box run -c /etc/sing-box/config.json
Restart=always
RestartSec=5
CapabilityBoundingSet=CAP_NET_ADMIN CAP_NET_RAW CAP_NET_BIND_SERVICE
AmbientCapabilities=CAP_NET_ADMIN CAP_NET_RAW CAP_NET_BIND_SERVICE

[Install]
WantedBy=multi-user.target
SSEOF

systemctl daemon-reload
systemctl enable sing-box
systemctl restart sing-box || warn "sing-box failed to start"

# ── 7. Firewall ────────────────────────────────────────────────────────────
log "Configuring firewall..."
# Allow SOCKS only from VPN
iptables -A INPUT -i wg0 -p tcp --dport ${SOCKS_PORT} -s 10.66.66.1 -j ACCEPT
iptables -A INPUT -p tcp --dport ${SOCKS_PORT} -j DROP

# IP forwarding
sysctl -w net.ipv4.ip_forward=1 > /dev/null
grep -q "net.ipv4.ip_forward=1" /etc/sysctl.conf || echo "net.ipv4.ip_forward=1" >> /etc/sysctl.conf

mkdir -p /etc/iptables
iptables-save > /etc/iptables/rules.v4 2>/dev/null || true

# ── 8. Store schedule (22:00-09:00 MSK) ────────────────────────────────────
log "Setting up store schedule..."
timedatectl set-timezone Europe/Moscow 2>/dev/null || \
  ln -sf /usr/share/zoneinfo/Europe/Moscow /etc/localtime

cat > /usr/local/bin/node-schedule.sh << 'SCHEDEOF'
#!/bin/bash
HOUR=$(date +%H)
if [[ $HOUR -ge 22 ]] || [[ $HOUR -lt 9 ]]; then
  systemctl start microsocks 2>/dev/null
else
  systemctl stop microsocks 2>/dev/null
fi
SCHEDEOF
chmod +x /usr/local/bin/node-schedule.sh

(crontab -l 2>/dev/null | grep -v node-schedule; \
 echo "*/15 * * * * /usr/local/bin/node-schedule.sh") | crontab -

/usr/local/bin/node-schedule.sh

# ── 9. Hostname ────────────────────────────────────────────────────────────
hostnamectl set-hostname "$NODE_NAME" 2>/dev/null || echo "$NODE_NAME" > /etc/hostname

# ── Summary ────────────────────────────────────────────────────────────────
echo ""
echo "======================================"
echo "  OPi2 Store Node — READY"
echo "======================================"
echo "  WG IP:        ${WG_IP}"
echo "  VPN tunnel:   wg0 → ${VPS_HOST}:${VPS_WG_PORT}"
echo "  SOCKS5:       0.0.0.0:${SOCKS_PORT}"
echo "  Reverse SSH:  VPS:${REVERSE_SSH_PORT} → localhost:22"
echo "  sing-box:     geo-routing active"
echo "  Schedule:     22:00-09:00 MSK"
echo "  Hostname:     ${NODE_NAME}"
echo "======================================"
echo ""
echo "Root SSH pubkey (add to VPS if not already there):"
cat /root/.ssh/id_ed25519.pub
echo ""
echo "Test commands:"
echo "  wg show wg0"
echo "  curl -x socks5://127.0.0.1:1080 ifconfig.me"
echo "  systemctl status autossh-tunnel"
echo "  systemctl status sing-box"
echo ""
